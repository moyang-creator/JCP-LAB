// 1. 更新 ABI 以匹配您的 V3 合约 const DEX_ABI = [ "function swapJCPForJCPUSD(uint256) external", "function swapJCPUSDForJCP(uint256) external", "function addLiquidity(uint256, uint256) external", "function removeLiquidity(uint256) external", "function getPoolStats() view returns (uint256, uint256, uint256, uint256)" ]; // 2. 修正后的数据刷新函数 async function updateUI() { if (!signer) return; const jcpAddr = document.getElementById('conf-jcp').value; const usdAddr = document.getElementById('conf-usd').value; const dexAddr = document.getElementById('conf-exchange').value; try { const jcp = new ethers.Contract(jcpAddr, ERC20_ABI, signer); const usd = new ethers.Contract(usdAddr, ERC20_ABI, signer); // 获取余额 const [bal, uBal] = await Promise.all([ jcp.balanceOf(userAddr), usd.balanceOf(userAddr) ]); document.getElementById('val-balance').innerText = ethers.utils.formatEther(bal); document.getElementById('val-usd-balance').innerText = ethers.utils.formatEther(uBal); // 获取池子状态 (使用 getPoolStats) try { const dex = new ethers.Contract(dexAddr, DEX_ABI, signer); const stats = await dex.getPoolStats(); // stats 分别对应: reserveJCP, reserveJCPUSD, totalLP, jcpPriceInUSD document.getElementById('pool-jcp').innerText = ethers.utils.formatEther(stats[0]); document.getElementById('pool-usd').innerText = ethers.utils.formatEther(stats[1]); // 价格直接使用合约计算好的值（注意合约放大 1e18） document.getElementById('val-price').innerText = ethers.utils.formatUnits(stats[3], 18); } catch (dexErr) { console.warn("读取池子失败，请检查 DEX 地址", dexErr); } } catch (e) { debugLog("❌ 同步失败：请确认合约地址正确", true); } } // 3. 修正后的兑换函数 async function executeSwap() { try { const dex = new ethers.Contract(document.getElementById('conf-exchange').value, DEX_ABI, signer); const amount = ethers.utils.parseEther(document.getElementById('swap-amount').value); const dir = document.getElementById('swap-direction').value; let tx; if (dir === 'JCPtoUSD') { tx = await dex.swapJCPForJCPUSD(amount); // 匹配 V3 合约名 } else { tx = await dex.swapJCPUSDForJCP(amount); // 匹配 V3 合约名 } debugLog("🔄 正在执行 V3 兑换..."); await tx.wait(); debugLog("✅ 兑换成功"); updateUI(); } catch (e) { debugLog("❌ 失败: " + (e.reason || "请检查余额及授权"), true); } }
